# ==================== CONFIGURACIÓN SERVIDOR ====================
PORT=5174
NODE_ENV=development

# Modo MOCK para pruebas sin LDAP (true = usar usuarios de prueba, false = usar LDAP real)
# Usuarios de prueba: testuser/password123, admin/admin123
MOCK_AUTH=true

# ==================== CONFIGURACIÓN LDAP/ACTIVE DIRECTORY ====================
# URL del servidor LDAP (usar ldaps:// para conexión segura)
LDAP_URL=ldaps://ad.tuempresa.com:636

# Base DN del dominio (ej: dc=tuempresa,dc=com)
LDAP_BASE_DN=dc=tuempresa,dc=com

# Formato de Distinguished Name para usuarios
# {username} será reemplazado por el nombre de usuario
# {baseDN} será reemplazado por LDAP_BASE_DN
LDAP_USER_DN_FORMAT=CN={username},CN=Users,{baseDN}

# Filtro de búsqueda LDAP para encontrar usuarios
# {username} será reemplazado por el nombre de usuario
LDAP_SEARCH_FILTER=(sAMAccountName={username})

# Cuenta técnica de servicio con permisos para autenticación y cambio de contraseñas
LDAP_ADMIN_DN=CN=svc_ldap_api,CN=Users,DC=tuempresa,DC=com
LDAP_ADMIN_PASSWORD=contraseña_segura_aqui

# Certificado CA (solo si el servidor LDAP usa una CA interna)
# Si el certificado está en un archivo, copia el contenido aquí (incluyendo BEGIN/END)
# LDAP_CA_CERT=-----BEGIN CERTIFICATE-----\n...

# Configuración de certificados SSL
# Si es false, no se verificarán certificados (solo para desarrollo)
LDAP_REJECT_UNAUTHORIZED=true

# Políticas de contraseña
LDAP_PASSWORD_MIN_LENGTH=8

# ==================== CONFIGURACIÓN JWT ====================
# Secreto para firmar tokens JWT (cambiar en producción)
JWT_SECRET=tu_secreto_jwt_super_seguro_cambiar_en_produccion

# Tiempo de expiración de tokens JWT (ej: 24h, 7d)
JWT_EXPIRES_IN=24h

# ==================== CONFIGURACIÓN THEREFORE ====================
THEREFORE_BASE_URL=https://tu-servidor.com/theservice/v0001/restun
THEREFORE_USERNAME=usuario_therefore
THEREFORE_PASSWORD=contraseña_therefore
THEREFORE_TENANT=nombre_tenant
